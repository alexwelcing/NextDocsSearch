class NeRF(nn.Module):
    def forward(self, position, direction):
        """
        Input: 3D position + viewing direction
        Output: RGB color + density
        """
        x = self.mlp(torch.cat([position, direction]))
        rgb = torch.sigmoid(x[:3])
        density = F.relu(x[3])
        return rgb, density
